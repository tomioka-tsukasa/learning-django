from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from rest_framework.permissions import AllowAny
from rest_framework.authtoken.models import Token
from rest_framework.request import Request
from .serializers import UserRegisterSerializer, LoginSerializer
from .types import LoginResponseData
import logging

logger = logging.getLogger(__name__)


class UserRegisterView(APIView):
    """ユーザー登録API"""

    permission_classes = [AllowAny]

    def post(self, request: Request) -> Response:
        """
        ユーザー登録

        Request Body:
            - username: ユーザー名（必須、3-150文字）
            - email: メールアドレス（必須）
            - password: パスワード（必須、Django標準バリデーション）
            - password_confirm: パスワード確認（必須）

        Returns:
            201: 登録成功
            400: バリデーションエラー
        """
        serializer = UserRegisterSerializer(data=request.data)

        if serializer.is_valid():
            user = serializer.save()
            logger.info(f"New user registered: {user.username}")

            return Response(
                {
                    "message": "ユーザー登録が完了しました。",
                    "user": {
                        "id": user.id,
                        "username": user.username,
                        "email": user.email
                    }
                },
                status=status.HTTP_201_CREATED
            )

        logger.warning(f"User registration failed: {serializer.errors}")
        return Response(
            {
                "message": "入力内容に誤りがあります。",
                "errors": serializer.errors
            },
            status=status.HTTP_400_BAD_REQUEST
        )


class LoginView(APIView):
    """ログインAPI（Token認証）"""

    permission_classes = [AllowAny]

    def post(self, request: Request) -> Response:
        """
        ログイン処理

        Request Body:
            - username: ユーザー名（必須）
            - password: パスワード（必須）

        Returns:
            200: ログイン成功、トークン返却
            400: バリデーションエラー
        """
        serializer = LoginSerializer(data=request.data)

        if serializer.is_valid():
            user = serializer.validated_data['user']

            # トークン取得または作成
            token, created = Token.objects.get_or_create(user=user)

            logger.info(f"User logged in: {user.username}")

            # 型安全なレスポンス
            response_data: LoginResponseData = {
                'token': token.key,
                'user': {
                    'id': user.id,
                    'username': user.username,
                    'email': user.email
                }
            }

            return Response(response_data, status=status.HTTP_200_OK)

        logger.warning(f"Login failed: {serializer.errors}")
        return Response(
            {
                "message": "ログインに失敗しました。",
                "errors": serializer.errors
            },
            status=status.HTTP_400_BAD_REQUEST
        )
